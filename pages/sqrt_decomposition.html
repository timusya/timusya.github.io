struct sqrt_decomposition {
 
    int n, len;
    vector<int> a, b;
    const int INF = 1e9;
 
    void init (int n) {
        this->n = n;
        len = (int)sqrt(n) + 1;
        a.resize(n, INF);
        b.resize(len, INF);
    }
 
    void build (const vector<int> &_a) {
        init(_a.size());
        a = _a;
        for (int i = 0; i < n; ++i) b[i / len] = min(b[i / len], a[i]);
    }
 
    int rmq (int l, int r) {
        int ans = INF;
        int c_l = l / len, c_r = r / len;
        if (c_l == c_r)
            for (int i = l; i <= r; ++i)
                ans = min(ans, a[i]);
        else {
            for (int i = l; i <= (c_l + 1) * len - 1; ++i)
                ans = min(ans, a[i]);
            for (int i = c_l + 1; i <= c_r - 1; ++i)
                ans = min(ans, b[i]);
            for (int i = c_r * len; i <= r; ++i)
                ans = min(ans, a[i]);
        }
        return ans;
    }
 
};
